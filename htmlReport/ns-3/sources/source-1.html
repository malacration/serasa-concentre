 package br.andrew.serasa.controllers
 
 import br.andrew.serasa.model.Consulta
 import br.andrew.serasa.repositorys.AcaoJudicialRepository
 import br.andrew.serasa.repositorys.ConsultaRepository
 import br.andrew.serasa.repositorys.PendenciaComercialRepository
 import br.andrew.serasa.services.SerasaService
 import org.springframework.web.bind.annotation.GetMapping
 import org.springframework.web.bind.annotation.PathVariable
 import org.springframework.web.bind.annotation.RequestMapping
 import org.springframework.web.bind.annotation.RestController
 
 
 @RestController
 @RequestMapping("serasa")
 class SerasaController(
     val repository : ConsultaRepository,
     val comercialRepository: PendenciaComercialRepository,
     val acaoRepository : AcaoJudicialRepository,
     val service : SerasaService) {
 
     @GetMapping("{cpf}")
     fun getTest(@PathVariable cpf : String): Any? {
         val parse = service.getByCpf(cpf)
         val consulta = repository.save(Consulta(parse))
         val acoes = parse.getAcaoJudicial().map { it.getAcaoJudicial(consulta) }
         val comercial = parse.getPendenciaComercial().map { it.getPendenciaComercial(consulta) }
         acaoRepository.saveAll(acoes)
         comercialRepository.saveAll(comercial)
         return parse
     }
 
     @GetMapping("{cpf}/score")
     fun score(@PathVariable cpf : String): Any? {
         return service.getByCpf(cpf,true)
     }
 }
